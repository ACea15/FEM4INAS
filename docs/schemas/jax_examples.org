  from diffrax import diffeqsolve, ODETerm, Dopri5
  import jax.numpy as jnp

  def f(t, y, args):
      return -y

  term = ODETerm(f)
  solver = Dopri5()
  y0 = jnp.array([2., 3.])
  solution = diffeqsolve(term, solver, t0=0, t1=1, dt0=0.1, y0=y0)
